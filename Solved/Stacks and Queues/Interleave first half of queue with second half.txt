#include <bits/stdc++.h> 
void interLeaveQueue(queue < int > & q) {
    int n=q.size()/2;
    queue<int>temp;
    for(int i=0;i<n;i++)
    {
        temp.push(q.front());
        q.pop();
    }

    while(!temp.empty())
    {
        q.push(temp.front());
        temp.pop();
        q.push(q.front());
        q.pop();
    }
}
